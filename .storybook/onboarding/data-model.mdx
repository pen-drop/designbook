# Data Model

Define the core entities and relationships in your product. The data model establishes the "nouns" of your system — what things exist and how they connect.
Use the AI command to define or update your data model — the result is displayed here automatically.

export function parseDataModel(md) {
  const entities = [];
  const relationships = [];
  const entitiesSection = md.match(/## Entities\s*\n+([\s\S]*?)(?=\n## |\n#[^#]|$)/);
  if (entitiesSection && entitiesSection[1]) {
    const entityMatches = [...entitiesSection[1].matchAll(/### ([^\n]+)\n+([\s\S]*?)(?=\n### |\n## |$)/g)];
    for (const match of entityMatches) {
      entities.push({
        name: match[1].trim(),
        description: match[2].trim(),
      });
    }
  }
  const relSection = md.match(/## Relationships\s*\n+([\s\S]*?)(?=\n## |\n#[^#]|$)/);
  if (relSection && relSection[1]) {
    const lines = relSection[1].split('\n');
    for (const line of lines) {
      const trimmed = line.trim();
      if (trimmed.startsWith('- ')) {
        relationships.push(trimmed.slice(2).trim());
      }
    }
  }
  if (entities.length === 0 && relationships.length === 0) return null;
  return { entities, relationships };
}
